12      /**
13       * Constructs {@link RandomMock} with given numbers that are used to give "random" numbers.
14       * @param numbers "random" numbers

18      }
19  
20      @Override
21      public int nextInt(int i) {
22          return nextInt();
23      }
24  
25      @Override
26      public int nextInt() {
27          int number = numbers[currentIndex++];
28          if (currentIndex >= numbers.length) {
29              currentIndex = 0;
30          }
31          return number;
32      }
