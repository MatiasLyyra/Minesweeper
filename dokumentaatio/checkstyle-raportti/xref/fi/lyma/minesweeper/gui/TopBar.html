10      public TopBar(MainWindow mainWindow) {
11          this.mainWindow = mainWindow;
12          JMenu file = new JMenu("File");
13          JMenuItem gameSettings = new JMenuItem("Change game settings");
14          gameSettings.addActionListener(listener -> mainWindow.showGameModeChooser());
15          file.add(gameSettings);
16          JMenuItem exit = new JMenuItem("Exit");
17          exit.addActionListener(listener -> System.exit(0));
18          file.add(gameSettings);
19          file.add(exit);
20          this.add(file);
21  
22      }
23
