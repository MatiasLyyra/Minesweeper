1   package fi.lyma.minesweeper.gui;
2   
3   import fi.lyma.minesweeper.logic.GameMode;
4   
5   import javax.swing.*;
6   import java.awt.*;
7   
8   /**
9    * Top bar that holds menu buttons.
10   */
11  public class TopBar extends JMenuBar {
12      private static final Dimension TOP_BAR_SIZE = new Dimension(256, 30);
13      private final MainWindow mainWindow;
14  
15      public TopBar(MainWindow mainWindow) {
16          this.mainWindow = mainWindow;
17          JMenu file = new JMenu("File");
18          JMenuItem gameSettings = new JMenuItem("Change game settings");
19          gameSettings.addActionListener(listener -> mainWindow.showGameModeChooser());
20          file.add(gameSettings);
21          JMenuItem exit = new JMenuItem("Exit");
22          exit.addActionListener(listener -> System.exit(0));
23          file.add(gameSettings);
24          file.add(exit);
25  
26          JMenu gameModes = new JMenu("Game Modes");
27          JMenuItem easy = new JMenuItem("Easy");
28          easy.addActionListener(listener -> mainWindow.startNewGame(GameMode.EASY));
29          JMenuItem medium = new JMenuItem("Medium");
30          medium.addActionListener(listener -> mainWindow.startNewGame(GameMode.MEDIUM));
31          JMenuItem hard = new JMenuItem("Hard");
32          hard.addActionListener(listener -> mainWindow.startNewGame(GameMode.HARD));
33          gameModes.add(easy);
34          gameModes.add(medium);
35          gameModes.add(hard);
36  
37          JMenu scores = new JMenu("Score");
38          JMenuItem highscores = new JMenuItem("High Scores");
39          highscores.addActionListener(listener -> mainWindow.showHighScores());
40          JMenuItem clearScore = new JMenuItem("Clear scores");
41          clearScore.addActionListener(listener -> mainWindow.clearScores());
42          scores.add(highscores);
43          scores.add(clearScore);
44  
45          this.add(file);
46          this.add(gameModes);
47          this.add(scores);
48      }
49  
50      @Override
51      public Dimension getPreferredSize() {
52          return TOP_BAR_SIZE;
53      }
54  }
