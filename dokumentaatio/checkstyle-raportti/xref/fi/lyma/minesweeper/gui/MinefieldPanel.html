3   import fi.lyma.minesweeper.logic.ImmutableTile;
4   import fi.lyma.minesweeper.logic.MinesweeperGame;
5   import fi.lyma.minesweeper.logic.Tile;
6   import fi.lyma.util.Vector2D;
7   
8   import javax.swing.*;
9   import java.awt.*;
10  import java.util.ArrayList;
11  import java.util.List;
12  
13  /**
14   * Draws minefield and highlights the tiles
15   */
16  public class MinefieldPanel extends JPanel {
17      private final MinesweeperGame game;
18      private List<ImmutableTile> highlightedTiles;
19      private final int tileSideLength = 32;
20  
21      public MinefieldPanel(MinesweeperGame game) {
22          highlightedTiles = new ArrayList<>();
23          this.game = game;
24      }
25  
26      @Override
27      public void paintComponent(Graphics graphics) {
28          super.paintComponent(graphics);
29          Graphics2D graphics2D = (Graphics2D) graphics;
30          for (int x = 0; x < game.getGameMode().getFieldWidth(); ++x) {
31              for (int y = 0; y < game.getGameMode().getFieldHeight(); ++y) {
32                  ImmutableTile tile = game.getTile(new Vector2D<>(x, y));
33                  graphics2D.drawImage(ImageResources.getImageWithTile(tile), x * tileSideLength, y * tileSideLength, null);
34              }
35          }
36          for (ImmutableTile highligted : highlightedTiles) {
37              graphics2D.drawImage(ImageResources.TILE_OPEN, highligted.getLocation().getX() * tileSideLength, highligted.getLocation().getY() * tileSideLength, null);
38          }
39      }
40  
41      @Override
42      public Dimension getPreferredSize() {
43          return new Dimension(tileSideLength * game.getGameMode().getFieldWidth(), tileSideLength * game.getGameMode().getFieldHeight());
44      }
45  
46      public Vector2D<Integer> convertScreenToWorldCoordinates(Vector2D<Integer> screenCoords) {
47          return new Vector2D<Integer>(screenCoords.getX() / tileSideLength, screenCoords.getY() / tileSideLength);
48      }
49  
50      public void clearHighlightedTile() {
51          highlightedTiles .clear();
52      }
53  
54      public void setHighlightedTile(Vector2D<Integer> location) {
55          location = convertScreenToWorldCoordinates(location);
56          ImmutableTile tile = game.getTile(location);
57          highlightedTiles.clear();
58          if (tile != null && tile.getStatus() != Tile.TileStatus.OPEN) {
59              highlightedTiles.add(tile);
60          }
61      }
62      public void highlightClosedAdjacentTiles(Vector2D<Integer> location) {
63          highlightedTiles.clear();
64          location = convertScreenToWorldCoordinates(location);
65          ImmutableTile tile = game.getTile(location);
66          if (tile != null && tile.getStatus() == Tile.TileStatus.OPEN) {
67              highlightedTiles.addAll(game.getAdjacentClosedNonFlaggedTiles(location));
68          }
69      }
70  
71  }
