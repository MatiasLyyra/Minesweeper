1   package fi.lyma.minesweeper.gui;
2   
3   import fi.lyma.minesweeper.logic.ImmutableTile;
4   import fi.lyma.minesweeper.logic.MinesweeperGame;
5   import fi.lyma.minesweeper.logic.Tile;
6   import fi.lyma.util.Vector2D;
7   
8   import javax.swing.*;
9   import java.awt.*;
10  
11  public class MinefieldPanel extends JPanel {
12      private final MinesweeperGame game;
13      private Vector2D<Integer> highlightedTile;
14      private final int tileSideLength = 32;
15  
16      public MinefieldPanel(MinesweeperGame game) {
17          this.game = game;
18      }
19  
20      @Override
21      public void paintComponent(Graphics graphics) {
22          super.paintComponent(graphics);
23          Graphics2D graphics2D = (Graphics2D) graphics;
24          for (int x = 0; x < game.getGameMode().getFieldWidth(); ++x) {
25              for (int y = 0; y < game.getGameMode().getFieldHeight(); ++y) {
26                  ImmutableTile tile = game.getTile(new Vector2D<>(x, y));
27                  graphics2D.drawImage(ImageResources.getImageWithTile(tile), x * tileSideLength, y * tileSideLength, null);
28              }
29          }
30          if (highlightedTile != null) {
31              graphics2D.drawImage(ImageResources.TILE_OPEN, highlightedTile.getX() * tileSideLength, highlightedTile.getY() * tileSideLength, null);
32          }
33      }
34  
35      @Override
36      public Dimension getPreferredSize() {
37          return new Dimension(tileSideLength * game.getGameMode().getFieldWidth(), tileSideLength * game.getGameMode().getFieldHeight());
38      }
39  
40      public Vector2D<Integer> convertScreenToWorldCoordinates(Vector2D<Integer> screenCoords) {
41          return new Vector2D<Integer>(screenCoords.getX() / tileSideLength, screenCoords.getY() / tileSideLength);
42      }
43  
44      public void clearHighlightedTile() {
45          highlightedTile = null;
46      }
47  
48      public void setHighlightedTile(Vector2D<Integer> location) {
49          location = convertScreenToWorldCoordinates(location);
50          ImmutableTile tile = game.getTile(location);
51          if (tile == null || tile.getStatus() == Tile.TileStatus.OPEN) {
52              highlightedTile = null;
53          } else {
54              highlightedTile = location;
55          }
56      }
57  }
