8   public class MainWindow implements Runnable {

10      private MinesweeperGame minesweeperGame;

14      public MainWindow(MinesweeperGame minesweeperGame) {

16          minefieldPanel = new MinefieldPanel(minesweeperGame);
17  
18      }
19  
20      @Override
21      public void run() {
22          frame = new JFrame();
23          frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
24          frame.setSize(600, 400);
25          frame.setResizable(true);
26          frame.add(minefieldPanel);
27          MinefieldMouseListener mouseListener = new MinefieldMouseListener(minefieldPanel, minesweeperGame);
28          minefieldPanel.addMouseListener(mouseListener);
29          minefieldPanel.addMouseMotionListener(mouseListener);
30          frame.setTitle("Minesweeper");
31          frame.pack();
32          frame.setVisible(true);
33      }
